SELECT FIRST_NAME, JOB_ID, SALARY,
DECODE(JOB_ID, 'IT_PROG',SALARY * 1.1,
             'FI_ACCOUNT', SALARY * 1.2,
             'AD_VP', SALARY * 0.95, SALARY)DECODE_CARGO
FROM HR.EMPLOYEES

CASE
WHEN SALARY < 5000 THEN 'AUMENTO'

'NÃO AUMENTAR'
END CLASSIFICAÇÃO
